<!-- View Counter -->
<div class="fixed bottom-4 left-4 z-40 glass px-3 py-2 rounded-lg shadow-lg text-sm hidden sm:flex items-center gap-2">
    <i class="fas fa-eye text-accent animate-pulse"></i>
    <span id="view-counter-gist">0</span>
    <span>views</span>
</div>

<script>
<script src="https://gist.github.com/Sarim2022/4457380f7abc02d15665a85b5d7568ca.js"></script>
const GIST_ID = 'YOUR_GIST_ID_HERE';
// Create a GitHub token with gist permission (optional but recommended)
const GITHUB_TOKEN = ''; // Leave empty for public read-only

async function updateViewCounter() {
    try {
        // First, get current count
        const response = await fetch(`https://api.github.com/gists/${GIST_ID}`);
        const data = await response.json();
        const currentContent = data.files['view_counter.json'].content;
        const currentViews = JSON.parse(currentContent).views;
        
        // Increment count
        const newViews = currentViews + 1;
        
        // Update display immediately
        document.getElementById('view-counter-gist').textContent = newViews;
        
        // Update Gist (requires token for write access)
        if (GITHUB_TOKEN) {
            await fetch(`https://api.github.com/gists/${GIST_ID}`, {
                method: 'PATCH',
                headers: {
                    'Authorization': `token ${GITHUB_TOKEN}`,
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({
                    files: {
                        'view_counter.json': {
                            content: JSON.stringify({ views: newViews })
                        }
                    }
                })
            });
        }
    } catch (error) {
        console.log('Using localStorage fallback');
        // Fallback to localStorage
        let count = parseInt(localStorage.getItem('gist_view_count')) || 0;
        count += 1;
        localStorage.setItem('gist_view_count', count);
        document.getElementById('view-counter-gist').textContent = count;
    }
}

// Run only once per session
if (!sessionStorage.getItem('viewCounted')) {
    sessionStorage.setItem('viewCounted', 'true');
    updateViewCounter();
} else {
    // Just show current count without incrementing
    document.getElementById('view-counter-gist').textContent = 
        localStorage.getItem('gist_view_count') || 'Loading...';
}
</script>
